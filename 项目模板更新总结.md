export async function getServerSideProps({ res }) {
  const sitemap = generateSiteMap(allScreens, staticPages);
  res.setHeader('Content-Type', 'text/xml');
  res.setHeader('Cache-Control', 'public, s-maxage=86400');

  res.end();
  return { props: {} };
}
